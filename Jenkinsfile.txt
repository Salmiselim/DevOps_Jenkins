fix it pipeline {
    agent any
    
    tools {
        maven 'M2_HOME'
    }
    
    environment {
        DOCKER_IMAGE = "selimsalmi/spring-app"
        DOCKER_TAG = "${BUILD_NUMBER}"
    }
    
    stages {
        stage('Checkout') {
            steps {
                git branch: 'master', url: 'https://github.com/Salmiselim/DevOps_Jenkins.git'
            }
        }
        
        stage('Build & Package') {
            steps {
                sh 'mvn clean install -DskipTests'
            }
        }
        
        stage('SonarQube Analysis') {
            steps {
                sh 'mvn sonar:sonar -Dsonar.host.url=http://192.168.10.7:9000 -Dsonar.token=squ_27398fb9a6fd885ad0ddf5f1c43c1840387ea0f2'
            }
        }
        
        stage('Build Docker Image') {
            steps {
                script {
                    sh "docker build -t ${DOCKER_IMAGE}:${DOCKER_TAG} -t ${DOCKER_IMAGE}:latest ."
                }
            }
        }
        
        stage('Push Docker Image') {
    steps {
        withCredentials([
            usernamePassword(
                credentialsId: 'docker-hub-credentials',
                usernameVariable: 'DOCKER_USERNAME',
                passwordVariable: 'DOCKER_PASSWORD'
            )
        ]) {
            sh '''
                echo "$DOCKER_PASSWORD" | docker login -u "$DOCKER_USERNAME" --password-stdin
                docker push --all-tags ${DOCKER_IMAGE}
            '''
        }
    }
}
        
        stage('Deploy Container') {
            steps {
                sh """
                    docker rm -f devops_container || true
                    docker run -d \
                        --name devops_container \
                        --network host \
                        ${DOCKER_IMAGE}:latest
                """
            }
        }
    }
    
    post {
        success {
            archiveArtifacts artifacts: 'target/*.jar', fingerprint: true, allowEmptyArchive: true
            echo " Pipeline completed successfully!"
            echo " Docker Image: ${DOCKER_IMAGE}:${DOCKER_TAG}"
        }
        failure {
            echo " Pipeline failed. Check logs for details."
        }
        always {
            sh 'docker logout || true'
        }
    }
}
